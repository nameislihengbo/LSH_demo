% 物理术语定义
% 版本: v5.0
% 更新日期: 2026-05-11

% ============================================
% 第一部分：波动与场
% ============================================

\def\WavePacketPhys{波包 (Wave Packet)}
\def\FieldPhys{场 (Field)}
\def\WavePhys{波 (Wave)}
\def\Frequency{f}                             % 频率
\def\Wavelength{\lambda}                       % 波长
\def\Amplitude{A}                              % 振幅
\def\Phase{\phi}                               % 相位

% ============================================
% 第二部分：能量与守恒
% ============================================

\def\KineticEnergy{E_{k}}                      % 动能
\def\PotentialEnergy{E_{p}}                    % 势能
\def\TotalEnergy{E_{total}}                    % 总能量
\def\EnergyFlux{\mathbf{S}}                     % 能流密度
\def\ConservationOfEnergy{能量守恒}

% ============================================
% 第三部分：算子与变换
% ============================================

\def\SelfAdjoint{自伴算子 (Self-Adjoint)}
\def\UnitaryOperator{酉算子 (Unitary Operator)}
\def\HermitianOperator{厄米算子 (Hermitian)}
\def\FourierTransform{\mathcal{F}}              % 傅里叶变换
\def\InverseFourier{\mathcal{F}^{-1}}          % 逆傅里叶变换

% ============================================
% 第四部分：因果与相对论
% ============================================

\def\LightCone{光锥 (Light Cone)}
\def\CausalStructure{因果结构}
\def\SpacetimeInterval{s^{2}}                   % 时空间隔
\def\ProperTime{\tau}                           % 固有时

% ============================================
% 第五部分：量子类比
% ============================================

\def\QuantumState{|\psi\rangle}                % 量子态
\def\WaveFunction{\psi(\mathbf{x},t)}          % 波函数
\def\SchrodingerEq{薛定谔方程}
\def\UncertaintyPrinciple{不确定性原理}
